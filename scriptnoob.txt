-- TP UNDER ENEMY FEATURE
-- =============================================
local TPUnderEnemy = {
    enabled = false,
    connection = nil,
    lockedY = nil,
    startPosition = nil,
    enemyTarget = nil,
    toggle = nil,
    keybind = nil
}

local function enableNoclip()
    local char = LocalPlayer.Character
    if not char then return end
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = false
        end
    end
end

local function disableNoclip()
    local char = LocalPlayer.Character
    if not char then return end
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = true
        end
    end
end

local function getClosestEnemy()
    local char = LocalPlayer.Character
    if not char then return nil end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return nil end
    
    local closest, minDist = nil, math.huge
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local dist = (root.Position - player.Character.HumanoidRootPart.Position).Magnitude
            if dist < minDist then
                minDist = dist
                closest = player
            end
        end
    end
    
    return closest
end

local function followEnemy()
    if not TPUnderEnemy.enabled or not TPUnderEnemy.lockedY or not TPUnderEnemy.enemyTarget then return end
    
    local char = LocalPlayer.Character
    if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return end
    
    local enemyChar = TPUnderEnemy.enemyTarget.Character
    if not enemyChar then
        TPUnderEnemy.enemyTarget = getClosestEnemy()
        return
    end
    
    local enemyRoot = enemyChar:FindFirstChild("HumanoidRootPart")
    if not enemyRoot then
        TPUnderEnemy.enemyTarget = getClosestEnemy()
        return
    end
    
    local enemyPos = enemyRoot.Position
    local followPos = Vector3.new(enemyPos.X, TPUnderEnemy.lockedY, enemyPos.Z)
    
    -- Smooth follow
    local currentPos = root.Position
    local lerpPos = currentPos:lerp(followPos, 0.3)
    root.CFrame = CFrame.new(lerpPos.X, TPUnderEnemy.lockedY, lerpPos.Z) * root.CFrame.Rotation
end

local function toggleTPUnderEnemy()
    if TPUnderEnemy.enabled then
        -- TURN OFF: Return to start position
        if TPUnderEnemy.startPosition then
            local char = LocalPlayer.Character
            if char then
                local root = char:FindFirstChild("HumanoidRootPart")
                if root then
                    root.CFrame = CFrame.new(TPUnderEnemy.startPosition) * root.CFrame.Rotation
                end
            end
        end
        
        disableNoclip()
        if TPUnderEnemy.connection then
            TPUnderEnemy.connection:Disconnect()
            TPUnderEnemy.connection = nil
        end
        
        TPUnderEnemy.enabled = false
        Library:Notify({Title = "TP Under", Description = "Disabled - Returned to start", Time = 3})
        
    else
        -- TURN ON: Save start pos, TP under enemy, lock Y
        local char = LocalPlayer.Character
        if not char then 
            Library:Notify({Title = "TP Under", Description = "No character found!", Time = 3})
            return
        end
        
        local root = char:FindFirstChild("HumanoidRootPart")
        if not root then
            Library:Notify({Title = "TP Under", Description = "No root part found!", Time = 3})
            return
        end
        
        TPUnderEnemy.startPosition = root.Position
        
        local enemy = getClosestEnemy()
        if not enemy or not enemy.Character or not enemy.Character:FindFirstChild("HumanoidRootPart") then
            Library:Notify({Title = "TP Under", Description = "No enemy found!", Time = 3})
            return
        end
        
        TPUnderEnemy.enemyTarget = enemy
        local enemyPos = enemy.Character.HumanoidRootPart.Position
        
        -- TP -7.5 under enemy
        local dropPos = Vector3.new(enemyPos.X, enemyPos.Y - 7.5, enemyPos.Z)
        root.CFrame = CFrame.new(dropPos) * root.CFrame.Rotation
        
        -- Lock Y at this height
        TPUnderEnemy.lockedY = dropPos.Y
        
        enableNoclip()
        TPUnderEnemy.connection = RunService.Heartbeat:Connect(followEnemy)
        
        TPUnderEnemy.enabled = true
        Library:Notify({Title = "TP Under", Description = "Enabled - Following enemy (Y locked)", Time = 3})
    end
    
    if TPUnderEnemy.toggle then
        TPUnderEnemy.toggle:Set(TPUnderEnemy.enabled)
    end
end

local function alive(p) 
    if not p or not p.Character then return false end
    local h = p.Character:FindFirstChildOfClass("Humanoid")
    return h and h.Health > 0
end

local function w2s(pos) 
    local v, on = Camera:WorldToViewportPoint(pos)
    return Vector2.new(v.X, v.Y), on, v.Z
end

local function dist3(a, b) 
    return (a - b).Magnitude
end

local function getPart(ch, name)
    return ch:FindFirstChild(name) or ch:FindFirstChild("Head") or ch:FindFirstChild("HumanoidRootPart")
end

local function getWeapon(ch)
    local t = ch:FindFirstChildOfClass("Tool")
    return t and t.Name or nil
end

local function getBones(ch)
    if ch:FindFirstChild("UpperTorso") then
        return {{"Head","UpperTorso"},{"UpperTorso","LowerTorso"},{"UpperTorso","RightUpperArm"},{"RightUpperArm","RightLowerArm"},{"RightLowerArm","RightHand"},{"UpperTorso","LeftUpperArm"},{"LeftUpperArm","LeftLowerArm"},{"LeftLowerArm","LeftHand"},{"LowerTorso","RightUpperLeg"},{"RightUpperLeg","RightLowerLeg"},{"RightLowerLeg","RightFoot"},{"LowerTorso","LeftUpperLeg"},{"LeftUpperLeg","LeftLowerLeg"},{"LeftLowerLeg","LeftFoot"}}
    elseif ch:FindFirstChild("Torso") then
        return {{"Head","Torso"},{"Torso","Right Arm"},{"Torso","Left Arm"},{"Torso","Right Leg"},{"Torso","Left Leg"}}
    end
    return {}
end

local function isVisible(from, to)
    local rp = RaycastParams.new()
    rp.FilterDescendantsInstances = {LocalPlayer.Character}
    rp.FilterType = Enum.RaycastFilterType.Exclude
    return workspace:Raycast(from, (to - from).Unit * dist3(from, to), rp) == nil
end

local function newDraw(cls, props)
    if not HasDrawing then return nil end
    local ok, d = pcall(function()
        local o = Drawing.new(cls)
        o.Visible = false
        for k, v in pairs(props or {}) do o[k] = v end
        return o
    end)
    return ok and d or nil
end

    FindSkinModel = function(self, skinName)
        if not skinName then return nil end
        for _, folder in pairs(self.skinFolders) do
            if folder then
                local success, model = pcall(function()
                    if folder:IsA("Folder") then
                        return folder:FindFirstChild(skinName)
                    else
                        return nil
                    end
                end)
                if success and model then
                    return model
                end
            end
        end
        return nil
    end,
    
    GetOrCreateBackupFolder = function(self)
        if self.backupFolder and self.backupFolder.Parent then
            return self.backupFolder
        end
        if not (self.viewModelsPath or self:InitializeViewModelsPath()) then
            return nil
        end
        local success, folder = pcall(function()
            local backup = self.viewModelsPath:FindFirstChild("OriginalBackups")
            if not backup then
                backup = Instance.new("Folder")
                backup.Name = "OriginalBackups"
                backup.Parent = self.viewModelsPath
            end
            return backup
        end)
        if not success then return nil end
        self.backupFolder = folder
        return folder
    end,
    
    ApplySkinModel = function(self, weaponName, skinName)
        if not (self.viewModelsPath or self:InitializeViewModelsPath()) then
            return false, "Couldn't find your weapons. Try restarting the game!"
        end
        local weaponModel = self.viewModelsPath:FindFirstChild(weaponName)
        if not (weaponModel and weaponModel:IsA("Model")) then
            return false, "Couldn't find the " .. weaponName .. " model. Try a different weapon!"
        end
        local skinModel = self:FindSkinModel(skinName)
        if not skinModel then
            return false, "Couldn't find the " .. skinName .. " skin. Try a different skin!"
        end
        local backupFolder = self:GetOrCreateBackupFolder()
        if not backupFolder then
            return false, "Couldn't create a backup. Try restarting the game!"
        end
        local backupName = weaponName .. "_Original"
        if not backupFolder:FindFirstChild(backupName) then
            local backup = weaponModel:Clone()
            backup.Name = backupName
            backup.Parent = backupFolder
        end
        weaponModel:Destroy()
        local newSkin = skinModel:Clone()
        newSkin.Name = weaponName
        newSkin.Parent = self.viewModelsPath
        return true, "Success! Your " .. weaponName .. " now has the " .. skinName .. " skin!"
    end,
    
    ResetSkinModel = function(self, weaponName)
        if not (self.viewModelsPath or self:InitializeViewModelsPath()) then
            return false, "Couldn't find your weapons. Try restarting the game!"
        end
        local backupFolder = self:GetOrCreateBackupFolder()
        if not backupFolder then
            return false, "Couldn't find your weapon backups. Try applying a skin first!"
        end
        local backup = backupFolder:FindFirstChild(weaponName .. "_Original")
        if not backup then
            return false, "No backup found for " .. weaponName .. ". Try applying a skin first!"
        end
        local currentWeapon = self.viewModelsPath:FindFirstChild(weaponName)
        if currentWeapon then
            currentWeapon:Destroy()
        end
        local restored = backup:Clone()
        restored.Name = weaponName
        restored.Parent = self.viewModelsPath
        return true, "Your " .. weaponName .. " is back to normal!"
    end
}

local function saveOriginalSkybox()
    local sky = Lighting:FindFirstChildOfClass("Sky")
    if sky then
        OriginalSkyboxes.SkyboxBk = sky.SkyboxBk
        OriginalSkyboxes.SkyboxDn = sky.SkyboxDn
        OriginalSkyboxes.SkyboxFt = sky.SkyboxFt
        OriginalSkyboxes.SkyboxLf = sky.SkyboxLf
        OriginalSkyboxes.SkyboxRt = sky.SkyboxRt
        OriginalSkyboxes.SkyboxUp = sky.SkyboxUp
    else
        OriginalSkyboxes.SkyboxBk = "rbxasset://sky/sky"
        OriginalSkyboxes.SkyboxDn = "rbxasset://sky/sky"
        OriginalSkyboxes.SkyboxFt = "rbxasset://sky/sky"
        OriginalSkyboxes.SkyboxLf = "rbxasset://sky/sky"
        OriginalSkyboxes.SkyboxRt = "rbxasset://sky/sky"
        OriginalSkyboxes.SkyboxUp = "rbxasset://sky/sky"
    end
end

local function forceSkyboxUpdate()
    if not SkyboxActive then return end
    
    local preset = SkyboxPresets[CurrentSkyboxIndex]
    if not preset then return end
    
    pcall(function()
        -- Remove existing custom skies
        for _, obj in ipairs(Lighting:GetChildren()) do
            if obj:IsA("Sky") and obj.Name == "CustomSky" then
                obj:Destroy()
            end
        end
        
        -- Create new sky
        local newSky = Instance.new("Sky")
        newSky.Name = "CustomSky"
        newSky.SkyboxBk = preset.IDs.SkyboxBk
        newSky.SkyboxDn = preset.IDs.SkyboxDn
        newSky.SkyboxFt = preset.IDs.SkyboxFt
        newSky.SkyboxLf = preset.IDs.SkyboxLf
        newSky.SkyboxRt = preset.IDs.SkyboxRt
        newSky.SkyboxUp = preset.IDs.SkyboxUp
        newSky.Parent = Lighting
        
        -- Also modify any existing sky
        for _, obj in ipairs(Lighting:GetChildren()) do
            if obj:IsA("Sky") and obj ~= newSky then
                obj.SkyboxBk = preset.IDs.SkyboxBk
                obj.SkyboxDn = preset.IDs.SkyboxDn
                obj.SkyboxFt = preset.IDs.SkyboxFt
                obj.SkyboxLf = preset.IDs.SkyboxLf
                obj.SkyboxRt = preset.IDs.SkyboxRt
                obj.SkyboxUp = preset.IDs.SkyboxUp
            end
        end
    end)
end

local function applySkybox(index)
    local preset = SkyboxPresets[index]
    if not preset then return end
    
    CurrentSkyboxIndex = index
    SkyboxActive = true
    forceSkyboxUpdate()
    
    -- Multiple force updates to ensure it works
    task.spawn(function()
        task.wait(0.1)
        forceSkyboxUpdate()
        task.wait(0.2)
        forceSkyboxUpdate()
    end)
    
    Library:Notify({Title = "Skybox", Description = "Applied: " .. preset.Name, Time = 3})
end

local function nextSkybox()
    if not Toggles.SkyboxOn.Value then return end
    
    CurrentSkyboxIndex = CurrentSkyboxIndex + 1
    if CurrentSkyboxIndex > #SkyboxPresets then
        CurrentSkyboxIndex = 1
    end
    
    applySkybox(CurrentSkyboxIndex)
    Options.SkyboxPreset:Set(SkyboxPresets[CurrentSkyboxIndex].Name)
end

local function restoreSkybox()
    SkyboxActive = false
    
    pcall(function()
        -- Remove custom skies
        for _, obj in ipairs(Lighting:GetChildren()) do
            if obj:IsA("Sky") and obj.Name == "CustomSky" then
                obj:Destroy()
            end
        end
        
        -- Restore original sky if exists
        if OriginalSkyboxes.SkyboxBk then
            local sky = Lighting:FindFirstChildOfClass("Sky")
            if not sky then
                sky = Instance.new("Sky")
                sky.Parent = Lighting
            end
            sky.SkyboxBk = OriginalSkyboxes.SkyboxBk
            sky.SkyboxDn = OriginalSkyboxes.SkyboxDn
            sky.SkyboxFt = OriginalSkyboxes.SkyboxFt
            sky.SkyboxLf = OriginalSkyboxes.SkyboxLf
            sky.SkyboxRt = OriginalSkyboxes.SkyboxRt
            sky.SkyboxUp = OriginalSkyboxes.SkyboxUp
        end
    end)
end

local DeviceSpooferLastSent = nil

local function findSetControlsRemote()
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local ok, obj = pcall(function()
        local remotes = ReplicatedStorage:FindFirstChild("Remotes")
        if not remotes then return nil end
        local replication = remotes:FindFirstChild("Replication")
        if not replication then return nil end
        local fighter = replication:FindFirstChild("Fighter")
        if not fighter then return nil end
        return fighter:FindFirstChild("SetControls")
    end)
    if ok then return obj end
    return nil
end

local function sendDeviceSpoofer()
    if not Toggles.DeviceSpooferOn or not Toggles.DeviceSpooferOn.Value then return end
    
    local deviceType = Options.DeviceSpooferType.Value
    if not deviceType then return end
    
    local MAP = {
        PC = "MouseKeyboard",
        Phone = "Touch",
        Controller = "Gamepad",
        VR = "VR",
    }
    
    local mapped = MAP[deviceType] or deviceType
    
    if DeviceSpooferRemoteCache == nil then
        DeviceSpooferRemoteCache = findSetControlsRemote()
    end
    
    if DeviceSpooferRemoteCache == nil then return end
    
    if DeviceSpooferLastSent == mapped then return end
    
    pcall(function()
        DeviceSpooferRemoteCache:FireServer(mapped)
    end)
    
    DeviceSpooferLastSent = mapped
end

local function fireBT()
    if not Toggles.BTOn.Value or not HasDrawing then return end
    pcall(function()
        local ch = LocalPlayer.Character
        if not ch then return end
        
        local barrel
        local tool = ch:FindFirstChildOfClass("Tool")
        if tool then
            local h = tool:FindFirstChild("Handle") or tool:FindFirstChildWhichIsA("BasePart")
            if h then barrel = h.Position end
        end
        
        if not barrel then
            local r = ch:FindFirstChild("RightHand") or ch:FindFirstChild("Right Arm")
            if r then barrel = r.Position end
        end
        
        if not barrel then
            local h = ch:FindFirstChild("Head")
            if h then barrel = h.Position end
        end
        
        if not barrel then return end
        
        local rp = RaycastParams.new()
        rp.FilterDescendantsInstances = {ch}
        rp.FilterType = Enum.RaycastFilterType.Exclude
        
        local res = workspace:Raycast(Camera.CFrame.Position, Camera.CFrame.LookVector * 1000, rp)
        local hitPos = res and res.Position or (Camera.CFrame.Position + Camera.CFrame.LookVector * 1000)
        
        local col = Options.BTCol.Value
        local typ = Options.BTType.Value
        local thick = Options.BTThick.Value
        
        if typ == "Default" then
            local l = newDraw("Line", {Thickness = thick, Color = col, Transparency = 0.1, Visible = true})
            if l then
                BT_Cache[#BT_Cache + 1] = {line = l, from3 = barrel, to3 = hitPos, birth = tick(), effect = "default"}
            end
        elseif typ == "Phantom" then
            for i = 1, 3 do
                local off = Vector3.new(math.random(-10, 10) / 10, math.random(-10, 10) / 10, math.random(-10, 10) / 10)
                local l = newDraw("Line", {Thickness = thick - (i * 0.3), Color = col, Transparency = 0.3 + (i * 0.2), Visible = true})
                if l then
                    BT_Cache[#BT_Cache + 1] = {line = l, from3 = barrel + off, to3 = hitPos + off, birth = tick(), effect = "phantom", age_offset = i * 0.1}
                end
            end
        elseif typ == "Spiral" then
            for i = 0, 9 do
                local pr = i / 10
                local a = pr * math.pi * 4
                local off = Vector3.new(math.cos(a) * 2, math.sin(a) * 2, 0)
                local l = newDraw("Line", {Thickness = thick, Color = col, Transparency = 0.1, Visible = true})
                if l then
                    BT_Cache[#BT_Cache + 1] = {line = l, from3 = barrel:Lerp(hitPos, pr) + off, to3 = barrel:Lerp(hitPos, pr + 0.1) + off, birth = tick(), effect = "spiral", age_offset = i * 0.05}
                end
            end
        elseif typ == "ZigZag" then
            for i = 0, 7 do
                local pr = i / 8
                local d2 = (i % 2 == 0) and 1 or -1
                local off = Vector3.new(d2 * 3, 0, d2 * 3)
                local l = newDraw("Line", {Thickness = thick, Color = col, Transparency = 0.1, Visible = true})
                if l then
                    BT_Cache[#BT_Cache + 1] = {line = l, from3 = barrel:Lerp(hitPos, pr) + off, to3 = barrel:Lerp(hitPos, pr + 0.15) + off, birth = tick(), effect = "zigzag", age_offset = i * 0.05}
                end
            end
        end
        
        spawnImpact(hitPos, col)
        
        if Toggles.BTGlow.Value then
            for i = 1, 3 do
                local g = Instance.new("Part")
                g.Size = Vector3.new(0.3, 0.3, 0.3)
                g.Shape = Enum.PartType.Ball
                g.Material = Enum.Material.Neon
                g.Anchored = true
                g.CanCollide = false
                g.Color = col
                g.Transparency = 0.2 + (i * 0.2)
                g.CFrame = CFrame.new(hitPos + Vector3.new(math.random(-5, 5) / 10, math.random(-5, 5) / 10, math.random(-5, 5) / 10))
                g.Parent = workspace
                TweenService:Create(g, TweenInfo.new(Options.BTLife.Value), {Transparency = 1, Size = Vector3.new(0.02, 0.02, 0.02)}):Play()
                Debris:AddItem(g, Options.BTLife.Value + 0.1)
            end
        end
        
        if Toggles.ShootFXOn.Value then
            spawnShootFX(barrel)
        end
    end)
end

local function updateBT()
    if not HasDrawing then return end
    
    if Toggles.BTOn.Value and BT_Mouse1Down then
        local now = tick()
        if now - BT_LastFire >= 0.05 then
            BT_LastFire = now
            fireBT()
        end
    end
    
    if not Toggles.BTOn.Value then
        for _, bt in ipairs(BT_Cache) do
            if bt.line then
                pcall(function() bt.line:Remove() end)
            end
        end
        BT_Cache = {}
        return
    end
    
    local now = tick()
    local life = Options.BTLife.Value
    
    if Toggles.BTRB.Value then
        BT_RBHue = (BT_RBHue + 0.005) % 1
    end
    
    local keep = {}
    for _, bt in ipairs(BT_Cache) do
        local age = now - bt.birth - (bt.age_offset or 0)
        if age < life and age > 0 then
            local fS, fOn = w2s(bt.from3)
            local tS, tOn = w2s(bt.to3)
            
            if bt.line and (fOn or tOn) then
                bt.line.From = fS
                bt.line.To = tS
                
                if Toggles.BTRB.Value then
                    bt.line.Color = Color3.fromHSV((BT_RBHue + age * 2) % 1, 1, 1)
                else
                    bt.line.Color = Options.BTCol.Value
                end
                
                bt.line.Transparency = Toggles.BTFade.Value and math.clamp(age / life, 0, 0.9) or 0.1
                bt.line.Visible = true
            elseif bt.line then
                bt.line.Visible = false
            end
            keep[#keep + 1] = bt
        else
            if bt.line then
                pcall(function() bt.line:Remove() end)
            end
        end
    end
    BT_Cache = keep
end

local function findTarget()
    local best, bestD = nil, Options.AimFOV.Value
    local ctr = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    local camP = Camera.CFrame.Position
    
    for _, p in ipairs(Players:GetPlayers()) do
        if p == LocalPlayer or not alive(p) then continue end
        
        local ch = p.Character
        if not ch then continue end
        
        local part = getPart(ch, Options.AimPart.Value)
        if not part then continue end
        
        if dist3(camP, part.Position) > Options.AimDist.Value then continue end
        
        if Toggles.AimVisCheck.Value and not isVisible(camP, part.Position) then continue end
        
        local sp, on = Camera:WorldToViewportPoint(part.Position)
        if on then
            local d = (Vector2.new(sp.X, sp.Y) - ctr).Magnitude
            if d < bestD then
                bestD = d
                best = p
            end
        end
    end
    
    return best
end

local function aimAt(tgt)
    if not tgt or not tgt.Character then return end
    
    local part = getPart(tgt.Character, Options.AimPart.Value)
    if not part then return end
    
    if Toggles.AimSticky.Value and StickyTarget ~= tgt then
        StickyTarget = tgt
        StickyLocked = true
    end
    
    if Toggles.AimHardLock.Value then
        -- Hard lock with CFrame (instant, 100% accurate)
        pcall(function()
            Camera.CFrame = CFrame.new(Camera.CFrame.Position, part.Position)
        end)
        return
    end
    
    -- MouseMoveRel aimbot with smoothing options
    local sp, on = Camera:WorldToViewportPoint(part.Position)
    if not on then return end
    
    local vp = Camera.ViewportSize
    local delta = Vector2.new(sp.X, sp.Y) - Vector2.new(vp.X / 2, vp.Y / 2)
    
    -- Apply smoothing based on selection
    local sm
    if Toggles.AimMouseHard.Value then
        -- Hard lock with mousemoverel (no smoothing)
        sm = 1
    else
        -- Smooth mousemoverel
        sm = Options.AimSm.Value
    end
    
    if Toggles.AimSticky.Value and StickyLocked then
        sm = sm * (1 - Options.AimStickyStr.Value)
    end
    
    sm = math.clamp(sm, 1, 20)
    mousemoverel(math.floor(delta.X / sm), math.floor(delta.Y / sm))
end

local CrossLines = {}
local CrossSpin = 0
local XRBHue = 0

local function initCrosshair()
    if not HasDrawing then return end
    
    for _, l in ipairs(CrossLines) do
        pcall(function() l:Remove() end)
    end
    CrossLines = {}
    
    for i = 1, 16 do
        CrossLines[i] = Drawing.new("Line")
        CrossLines[i].Visible = false
        CrossLines[i].Thickness = 2
    end
end

local function updateCrosshair()
    if not HasDrawing or not Toggles.XOn.Value then
        for _, l in ipairs(CrossLines) do
            if l then l.Visible = false end
        end
        return
    end
    
    local vp = Camera.ViewportSize
    local mx, my = vp.X / 2, vp.Y / 2
    local size, gap, thick = Options.XSize.Value, Options.XGap.Value, Options.XThick.Value
    local hasDot = Toggles.XDot.Value
    
    CrossSpin = CrossSpin + math.rad(Options.XSpin.Value)
    
    local color = Options.XCol.Value
    if Toggles.XRainbow.Value then
        XRBHue = (XRBHue + 0.005) % 1
        color = Color3.fromHSV(XRBHue, 1, 1)
    end
    
    local glow = Toggles.XGlow.Value
    local cosA, sinA = math.cos(CrossSpin), math.sin(CrossSpin)
    
    local function rot(px, py)
        local dx, dy = px - mx, py - my
        return mx + dx * cosA - dy * sinA, my + dx * sinA + dy * cosA
    end
    
    for i = 1, 16 do
        if CrossLines[i] then
            CrossLines[i].Visible = false
        end
    end
    
    local style = Options.XStyle.Value
    local lineIndex = 1
    
    if style == "Cross" then
        local pts = {
            {mx, my - gap, mx, my - gap - size},
            {mx, my + gap, mx, my + gap + size},
            {mx - gap, my, mx - gap - size, my},
            {mx + gap, my, mx + gap + size, my}
        }
        for i = 1, 4 do
            local l = CrossLines[lineIndex]
            if l then
                local x1, y1 = rot(pts[i][1], pts[i][2])
                local x2, y2 = rot(pts[i][3], pts[i][4])
                l.From = Vector2.new(x1, y1)
                l.To = Vector2.new(x2, y2)
                l.Color = color
                l.Thickness = thick + (glow and 1 or 0)
                l.Transparency = glow and 0.5 or 0
                l.Visible = true
                lineIndex = lineIndex + 1
            end
        end
    elseif style == "Circle" then
        for i = 1, 8 do
            local l = CrossLines[lineIndex]
            if l then
                local a1 = (i - 1) * (math.pi * 2 / 8)
                local a2 = i * (math.pi * 2 / 8)
                l.From = Vector2.new(mx + math.cos(a1) * size, my + math.sin(a1) * size)
                l.To = Vector2.new(mx + math.cos(a2) * size, my + math.sin(a2) * size)
                l.Color = color
                l.Thickness = thick + (glow and 1 or 0)
                l.Transparency = glow and 0.5 or 0
                l.Visible = true
                lineIndex = lineIndex + 1
            end
        end
    end
    
    if hasDot then
        local dotSize = size * 0.15
        if dotSize < 2 then dotSize = 2 end
        local l1 = CrossLines[lineIndex]
        local l2 = CrossLines[lineIndex + 1]
        if l1 and l2 then
            l1.From = Vector2.new(mx - dotSize, my)
            l1.To = Vector2.new(mx + dotSize, my)
            l1.Color = color
            l1.Thickness = thick
            l1.Transparency = 0
            l1.Visible = true
            
            l2.From = Vector2.new(mx, my - dotSize)
            l2.To = Vector2.new(mx, my + dotSize)
            l2.Color = color
            l2.Thickness = thick
            l2.Transparency = 0
            l2.Visible = true
        end
    end
end

local function removeCache(p)
    local c = ESPCache[p]
    if not c then return end
    
    for _, o in ipairs(c._a) do
        pcall(function() o:Remove() end)
    end
    
    ESPCache[p] = nil
end

local function hideAll(p)
    local c = ESPCache[p]
    if not c then return end
    
    for _, o in ipairs(c._a) do
        pcall(function() o.Visible = false end)
    end
end

local function updateDrawESP(player)
    local c = ESPCache[player]
    if not c or not Toggles.ESPOn.Value then
        if c then hideAll(player) end
        return
    end
    
    local ch = player.Character
    if not ch or not alive(player) then
        hideAll(player)
        return
    end
    
    local hum = ch:FindFirstChildOfClass("Humanoid")
    local root = ch:FindFirstChild("HumanoidRootPart") or ch:FindFirstChild("Torso")
    local head = ch:FindFirstChild("Head")
    
    if not root or not head or not hum then
        hideAll(player)
        return
    end
    
    local camP = Camera.CFrame.Position
    local d = dist3(camP, root.Position)
    
    if d > Options.ESPMaxD.Value then
        hideAll(player)
        return
    end
    
    local rS, rOn = w2s(root.Position)
    if not rOn then
        hideAll(player)
        return
    end
    
    local tS, tOn = w2s(head.Position + Vector3.new(0, head.Size.Y / 2 + 0.3, 0))
    local bS, bOn = w2s(root.Position - Vector3.new(0, 3, 0))
    
    if not tOn or not bOn then
        hideAll(player)
        return
    end
    
    local bH = math.abs(bS.Y - tS.Y)
    if bH < 2 then
        hideAll(player)
        return
    end
    
    local bW = bH * 0.6
    local cx = (tS.X + bS.X) / 2

    local tr = Vector2.new(cx + bW / 2, tS.Y)

            c.wp.Visible = false
        end
    elseif c.wp then
        c.wp.Visible = false
    end
    
    if Toggles.ESPHP.Value and hum then
        local pct = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
        local bw2, bx = 3, tl.X - 7
        
        if c.hBo then
            c.hBo.Position = Vector2.new(bx - 1, tS.Y - 1)
            c.hBo.Size = Vector2.new(bw2 + 2, bH + 2)
            c.hBo.Visible = true
        end
        
        if c.hBg then
            c.hBg.Position = Vector2.new(bx, tS.Y)
            c.hBg.Size = Vector2.new(bw2, bH)
            c.hBg.Visible = true
        end
        
        if c.hFi then
            local fh = bH * pct
            c.hFi.Position = Vector2.new(bx, tS.Y + (bH - fh))
            c.hFi.Size = Vector2.new(bw2, fh)
            c.hFi.Color = Options.ESPHPCol.Value
            c.hFi.Visible = true
        end
    else
        if c.hBg then c.hBg.Visible = false end
        if c.hBo then c.hBo.Visible = false end
        if c.hFi then c.hFi.Visible = false end
    end
    
    if Toggles.ESPTracer.Value then
        local vp = Camera.ViewportSize
        local from
        local o = Options.ESPTracFrom.Value
        
        if o == "Top" then
            from = Vector2.new(vp.X / 2, 0)
        elseif o == "Middle" then
            from = Vector2.new(vp.X / 2, vp.Y / 2)
        else
            from = Vector2.new(vp.X / 2, vp.Y)
        end
        
        if c.trO then
            c.trO.From = from
            c.trO.To = bS
            c.trO.Visible = true
        end
        
        if c.tr then
            c.tr.From = from
            c.tr.To = bS
            c.tr.Color = Options.ESPTracCol.Value
            c.tr.Visible = true
        end
    else
        if c.tr then c.tr.Visible = false end
        if c.trO then c.trO.Visible = false end
    end
end

-- HIGHLIGHT FUNCTIONS
-- =============================================
local function addHL(p)
    if p == LocalPlayer then return end
    
    local function onChar(ch)
        pcall(function()
            if Highlights[p] then Highlights[p]:Destroy() end
        end)
        
        local hl = Instance.new("Highlight")
        hl.Name = "OHL"
        hl.FillTransparency = 0.75
        hl.OutlineTransparency = 0
        hl.Adornee = ch
        hl.Enabled = false
        hl.Parent = ch
        Highlights[p] = hl
        
        if HasDrawing then
            makeCache(p)
        end
    end
    
    if p.Character then
        onChar(p.Character)
    end
    
    p.CharacterAdded:Connect(onChar)
end

local function removeHL(p)
    pcall(function()
        if Highlights[p] then
            Highlights[p]:Destroy()
        end
    end)
    
    Highlights[p] = nil
    removeCache(p)
    EnemyHP[p] = nil
    PrevAliveState[p] = nil
end

for _, p in ipairs(Players:GetPlayers()) do
    addHL(p)
end

Players.PlayerAdded:Connect(addHL)
Players.PlayerRemoving:Connect(removeHL)

ChangerLeft:AddLabel("Find Your Favorite Skins", true)

ChangerLeft:AddInput("SearchInput", {
    Default = "",
    Numeric = false,
    Finished = false,
    Text = "Search Weapons and Skins",
    Tooltip = "Enter a weapon or skin name to filter options",
    Placeholder = "Example: AK-47, Knife, Pixel...",
    Callback = function(searchText)
        local search = searchText:lower()
        
        if search == "" then
            -- Reset to full lists
            Options.WeaponSelect:SetValues(WeaponList)
            Options.WeaponSelect:SetValue(CurrentWeapon)
            
            local skins = WeaponData[CurrentWeapon] or {}
            if #skins <= 0 then
                Options.SkinSelect:SetValues({"No Skins Found"})
                Options.SkinSelect:SetValue("No Skins Found")
            else
                Options.SkinSelect:SetValues(skins)
                Options.SkinSelect:SetValue(CurrentSkin)
            end
        else
            -- Search weapons
            local matchedWeapons = {}
            for _, weapon in ipairs(WeaponList) do
                if weapon:lower():find(search) then
                    table.insert(matchedWeapons, weapon)
                end
            end
            
            -- Search skins and track their weapons
            local weaponsWithSkinMatches = {}
            local skinMatches = {}
            
            for weapon, skins in pairs(WeaponData) do
                local matchedSkins = {}
                for _, skin in ipairs(skins) do
                    if skin:lower():find(search) then
                        table.insert(matchedSkins, skin)
                        if not table.find(weaponsWithSkinMatches, weapon) then
                            table.insert(weaponsWithSkinMatches, weapon)
                        end
                    end
                end
                if #matchedSkins > 0 then
                    skinMatches[weapon] = matchedSkins
                end
            end
            
            -- Combine results
            local results = {}
            for _, weapon in ipairs(matchedWeapons) do
                if not table.find(results, weapon) then
                    table.insert(results, weapon)
                end
            end
            for _, weapon in ipairs(weaponsWithSkinMatches) do
                if not table.find(results, weapon) then
                    table.insert(results, weapon)
                end
            end
            
            if #results <= 0 then
                Options.WeaponSelect:SetValues({"No Matches Found"})
                Options.WeaponSelect:SetValue("No Matches Found")
                Options.SkinSelect:SetValues({"No Matches Found"})
                Options.SkinSelect:SetValue("No Matches Found")
                CurrentWeapon = "No Matches Found"
                CurrentSkin = "No Matches Found"
            else
                Options.WeaponSelect:SetValues(results)
                
                if table.find(results, CurrentWeapon) then
                    if skinMatches[CurrentWeapon] then
                        Options.SkinSelect:SetValues(skinMatches[CurrentWeapon])
                        CurrentSkin = skinMatches[CurrentWeapon][1]
                        Options.SkinSelect:SetValue(CurrentSkin)
                    end
                else
                    CurrentWeapon = results[1]
                    Options.WeaponSelect:SetValue(CurrentWeapon)
                    
                    if skinMatches[CurrentWeapon] then
                        Options.SkinSelect:SetValues(skinMatches[CurrentWeapon])
                        CurrentSkin = skinMatches[CurrentWeapon][1]
                        Options.SkinSelect:SetValue(CurrentSkin)
                    else
                        local skins = WeaponData[CurrentWeapon] or {"No Skins Found"}
                        Options.SkinSelect:SetValues(skins)
                        CurrentSkin = skins[1]
                        Options.SkinSelect:SetValue(CurrentSkin)
                    end
                end
            end
        end
    end
})

RunService.Heartbeat:Connect(function()
    -- FOV Circle Update
    if HasDrawing and FOVDraw and FOVFillDraw and Toggles.AimShowFOV.Value then
        local vp = Camera.ViewportSize
        local ctr = Vector2.new(vp.X / 2, vp.Y / 2)
        local fov = Options.AimFOV.Value
        
        FOVDraw.Radius = fov
        FOVDraw.Position = ctr
        FOVDraw.Color = Options.AimFOVCol.Value
        FOVDraw.Transparency = Options.AimFOVTrans.Value
        FOVDraw.Visible = true
        
        FOVFillDraw.Radius = fov
        FOVFillDraw.Position = ctr
        FOVFillDraw.Color = Options.AimFOVFillCol.Value
        FOVFillDraw.Transparency = Options.AimFOVFillTrans.Value
        FOVFillDraw.Visible = Toggles.AimFOVFill.Value
    elseif FOVDraw then
        FOVDraw.Visible = false
        if FOVFillDraw then FOVFillDraw.Visible = false end
    end
    
    -- Aimbot
    if Toggles.AimOn.Value and Options.AimKey:GetState() then
        local target = StickyLocked and StickyTarget or findTarget()
        if target then
            aimAt(target)
        else
            StickyLocked = false
            StickyTarget = nil
        end
    else
        StickyLocked = false
        StickyTarget = nil
    end
    
    -- Scope
    if Toggles.ScopeOn.Value then
        if Options.ScopeKey:GetState() then
            startScope()
        else
            stopScope()
        end
    end
    
    -- Rainbow ESP hue update
    if Toggles.ESPRainbow.Value then
        RainbowHue = (RainbowHue + (Options.ESPRBSpeed.Value / 1000)) % 1
    end
    
    -- Highlights + Drawing ESP
    for p, hl in pairs(Highlights) do
        if hl and hl.Parent then
            hl.Enabled = Toggles.ESPOn.Value and Toggles.ESPHL.Value
            
            if Toggles.ESPRainbow.Value then
                hl.FillColor = Color3.fromHSV(RainbowHue, 1, 1)
                hl.OutlineColor = Color3.fromHSV(RainbowHue, 1, 1)
            else
                hl.FillColor = Options.ESPHLCol.Value
                hl.OutlineColor = Color3.new(1, 1, 1)
            end
            
            hl.FillTransparency = Options.ESPFillT.Value
            hl.OutlineTransparency = Options.ESPOutT.Value
        end
    end
    
    if HasDrawing then
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer then
                if not ESPCache[p] and p.Character then
                    makeCache(p)
                end
                updateDrawESP(p)
            end
        end
    end
    
    updateCrosshair()
    updateBT()
    updateSpeed()
    
    -- Device Spoofer Heartbeat
    if Toggles.DeviceSpooferOn and Toggles.DeviceSpooferOn.Value then
        sendDeviceSpoofer()
    end
    
    if Toggles.Noclip.Value and not TPUnderEnemy.enabled then
        pcall(function()
            local ch = LocalPlayer.Character
            if ch then
                for _, p in ipairs(ch:GetDescendants()) do
                    if p:IsA("BasePart") then
                        p.CanCollide = false
                    end
                end
            end
        end)
    end
    
    -- Apply lighting overrides
    if Toggles.FullBright.Value then
        Lighting.Ambient = Color3.new(1, 1, 1)
        Lighting.Brightness = 2
        Lighting.OutdoorAmbient = Color3.new(1, 1, 1)
    elseif not Toggles.AmbienceOn.Value or not CurrentAmbience then
        Lighting.Ambient = OrigLighting.Ambient
        Lighting.Brightness = OrigLighting.Brightness
        Lighting.OutdoorAmbient = OrigLighting.OutdoorAmbient
    end
    
    if Toggles.NoFog.Value then
        Lighting.FogEnd = 1e9
    elseif not Toggles.AmbienceOn.Value or not CurrentAmbience then
        Lighting.FogEnd = OrigLighting.FogEnd
    end
    
    if Toggles.NoShadows.Value then
        Lighting.GlobalShadows = false
    elseif not Toggles.AmbienceOn.Value or not CurrentAmbience then
        Lighting.GlobalShadows = OrigLighting.GlobalShadows
    end
    
    if Toggles.ForceDay.Value and not (Toggles.AmbienceOn.Value and CurrentAmbience) then
        Lighting.ClockTime = 14
    end
    
    if Toggles.ForceNight.Value and not (Toggles.AmbienceOn.Value and CurrentAmbience) then
        Lighting.ClockTime = 0
    end
    
    -- Apply color correction and bloom
    pcall(function()
        if not ColorCorrection then
            ColorCorrection = Instance.new("ColorCorrectionEffect", Lighting)
            ColorCorrection.Name = "OCC"
        end
        ColorCorrection.Saturation = Options.ColorSat.Value
        ColorCorrection.Contrast = Options.ColorCon.Value * 2
        ColorCorrection.Brightness = Options.ColorBri.Value - 1
    end)
    
    pcall(function()
        if Options.BloomInt.Value > 0 then
            if not BloomEffect then
                BloomEffect = Instance.new("BloomEffect", Lighting)
                BloomEffect.Name = "OBL"
            end
            BloomEffect.Intensity = Options.BloomInt.Value
            BloomEffect.Threshold = Options.BloomThr.Value
            BloomEffect.Size = Options.BloomSz.Value
        elseif BloomEffect then
            BloomEffect:Destroy()
            BloomEffect = nil
        end
    end)
    
    -- Hit/Kill Sound Tracking
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character then
            local ph = p.Character:FindFirstChildOfClass("Humanoid")
            if ph then
                local prev = EnemyHP[p]
                local cur = ph.Health
                
                -- Hit sound
                if prev and cur < prev and (prev - cur) < 100 and (prev - cur) > 0 then
                    if Toggles.HitSndOn.Value then
                        playSound(getHitSoundID(Options.HitSndPreset.Value), Options.HitSndVol.Value)
                    end
                end
                
                EnemyHP[p] = cur
                
                -- Kill sound
                local wasAlive = PrevAliveState[p]
                local isAlive = ph.Health > 0
                
                if wasAlive == true and not isAlive then
                    if Toggles.KillSndOn.Value then
                        playSound(getKillSoundID(Options.KillSndPreset.Value), Options.KillSndVol.Value)
                    end
                end
                
                PrevAliveState[p] = isAlive
            end
        end
    end
end)

Library:OnUnload(function()
    for _, c in pairs(Connections) do
        if c and typeof(c) == "RBXScriptConnection" then
            c:Disconnect()
        end
    end
    
    if TPUnderEnemy.connection then
        TPUnderEnemy.connection:Disconnect()
    end
    
    if SkyboxForceConnection then
        SkyboxForceConnection:Disconnect()
    end
    
    if HasDrawing then
        if FOVDraw then pcall(function() FOVDraw:Remove() end) end
        if FOVFillDraw then pcall(function() FOVFillDraw:Remove() end) end
        for _, l in ipairs(CrossLines) do pcall(function() l:Remove() end) end
        for _, bt in ipairs(BT_Cache) do
            if bt.line then
                pcall(function() bt.line:Remove() end)
            end
        end
    end
    
    for p, _ in pairs(ESPCache) do removeCache(p) end
    for _, hl in pairs(Highlights) do pcall(function() if hl then hl:Destroy() end end) end
    
    restoreSkybox()
    restoreAmbience()
    
    for o, e in pairs(StoredEffects) do
        if o and o.Parent then
            o.Enabled = e
        end
    end
    
    if BloomEffect then pcall(function() BloomEffect:Destroy() end) end
    if ColorCorrection then pcall(function() ColorCorrection:Destroy() end) end
    
    Camera.FieldOfView = 70
    
    if TPUnderEnemy.enabled then
        if TPUnderEnemy.startPosition then
            local char = LocalPlayer.Character
            if char then
                local root = char:FindFirstChild("HumanoidRootPart")
                if root then
                    root.CFrame = CFrame.new(TPUnderEnemy.startPosition) * root.CFrame.Rotation
                end
            end
        end
        disableNoclip()
    end
end)
